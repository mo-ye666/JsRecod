#### 实现简单proxy双向绑定
```js
    const data = {value : ''};
    const handle = {
        get(target,property){
            return target[property]
        },
        set(target,property,value){
            target[property] = value;
            upDateView(value);
            return true
        }
    };
    const proxy = new Proxy(data,handle);
    function upDateView(value) {
        console.log('view updated:',value)
    }
    function upDateData(event) {
        proxy.data = event.traget.value;
    }
    const inputElement = document.getElementById(input);
    inputElement.addEventListener('input',upDateData);
```

Function.prototype.call = function(context,args){
    if(context === undefined || context === null) {
        context = window
    }
    let fnSymbo = Symbol();
    context[fnSymbo] = this;
    let fn = context[fnSymbo](...args);
    delete context[fnSymbo]
    return context;
}
